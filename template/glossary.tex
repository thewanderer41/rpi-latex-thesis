\newglossaryentry{version}{
	name={version},
	description={An \textbf{expression} of a \textbf{work} which exists in comparison to another object and communicates the extent to which it diverges from that object as a result of provenance changes}
}

\newglossaryentry{log}{
	name={change log},
	description={A text document describing the changes between two \glspl{version}}
}

\newglossaryentry{attribute}{
	name={attribute},
	description={Part of a \gls{version} used to determine if a \gls{version} has changed}
}

\newglossaryentry{change}{
	name={change},
	description={A relationship between a \gls{version} and an \gls{attribute} or between two \glspl{attribute} indicating a difference between the entities}
}

\newglossaryentry{add}{
	name={addition},
	description={A \gls{version}-to-\gls{attribute} \gls{change} indicating the \gls{attribute} has been added to the \gls{version}}
}

\newglossaryentry{invalidate}{
	name={invalidation},
	description={An \gls{attribute}-to-\gls{version} \gls{change} indicating the \gls{attribute} no longer appears in the \gls{version}}
}

\newglossaryentry{modify}{
	name={modification},
	description={An \gls{attribute}-to-\gls{attribute} \gls{change} indicating values of the \gls{attribute} differ between \glspl{version}},
	plural={modifies}
}

\newglossaryentry{vergraph}{
	name={versioning graph},
	description={Instantiation of a \gls{version} transition or series of transitions using the Versioning Ontology}
}

\newglossaryentry{provenance}{
	name={provenance},
	description={Information about entities, activities, and people involved in producing a piece of data or thing, which can be used to form assessments about its quality, reliability or trustworthiness \cite{Moreau2013c}}}

\newglossaryentry{provdist}{
	name={provenance distance},
	description={A measure of difference between two objects based on their provenance graphs}
}

\newglossaryentry{changedist}{
	name={change distance},
	description={A difference score between two objects composed of three components, each counting one of the \acrshort{AIM} changes}
}

\newglossaryentry{linked}{
	name={linked-data},
	description={The collection of interrelated datasets on the Web of Data \cite{ld}},
	plural={linked data}
}

\newglossaryentry{volatility}{
	name={volatility},
	description={the likelihood of data change}
}
